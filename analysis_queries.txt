--> 1. Total Sales Revenue:
SELECT SUM(quantity * price) AS total_revenue
FROM sales;

-->2. Total Sales by Product
SELECT product_name,
SUM(quantity) AS total_quantity
FROM sales
GROUP BY product_name;

--> 3. Revenue by Category
SELECT category,
SUM(quantity * price) AS revenue
FROM sales
GROUP BY category;

--> 4. Top Selling Product
SELECT product_name,
SUM(quantity) AS total_sold
FROM sales
GROUP BY product_name
ORDER BY total_sold DESC
LIMIT 1;

--> 5. Monthly Sales
SELECT MONTH(sale_date) AS month,
SUM(quantity * price) AS revenue
FROM sales
GROUP BY MONTH(sale_date);


 Total Sales Revenue → Calculates overall revenue
 Sales by Product → Shows quantity sold for each product
 Revenue by Category → Shows revenue generated by each category
 Top Selling Product → Finds the product with highest sales
 Monthly Sales → Shows revenue for each month
